import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns  # Import Seaborn

# Load the CSV file into a Pandas DataFrame
df = pd.read_csv("public transport analysis efficiency.csv")

# Calculate Total Boardings by Stop
total_boardings_by_stop = df.groupby('StopName')['NumberOfBoardings'].sum()

# Visualize the top 10 stops with the most boardings
top_10_stops = total_boardings_by_stop.nlargest(10)

# Enhance the plot with Seaborn
sns.set(style="whitegrid")  # Set a Seaborn style
plt.figure(figsize=(10, 6))  # Set the figure size
ax = sns.barplot(x=top_10_stops.index, y=top_10_stops.values)
ax.set(title='Top 10 Stops by Total Boardings', xlabel='Stop Name', ylabel='Total Boardings')

# Show the plot
plt.xticks(rotation=45)  # Rotate x-axis labels for better readability
plt.show()

# Calculate Average Boardings per Week for each Route
average_boardings_by_route = df.groupby('RouteID')['NumberOfBoardings'].mean()

# Visualize average boardings for each route
plt.figure(figsize=(10, 6))  # Set the figure size
ax = sns.barplot(x=average_boardings_by_route.index, y=average_boardings_by_route.values)
ax.set(title='Average Boardings per Week by Route', xlabel='Route ID', ylabel='Average Boardings per Week')

# Show the plot
plt.xticks(rotation=45)
plt.show()

# Analyze trends over time
df['WeekBeginning'] = pd.to_datetime(df['WeekBeginning'])
boardings_over_time = df.groupby('WeekBeginning')['NumberOfBoardings'].sum()

# Visualize boardings trend over time
plt.figure(figsize=(12, 6))  # Set the figure size
ax = sns.lineplot(x=boardings_over_time.index, y=boardings_over_time.values)
ax.set(title='Total Boardings Trend Over Time', xlabel='Week Beginning', ylabel='Total Boardings')

# Show the plot
plt.xticks(rotation=45)
plt.show()
